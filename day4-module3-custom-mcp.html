Setting Up the Python MCP SDK

The mcp[cli] Python package provides everything you need to build a production MCP server. The SDK handles the JSON-RPC 2.0 transport, capability negotiation, and message routing — you write only the business logic for your tools, resources, and prompts.

bash

pip install "mcp[cli]"

    articles.py       # get_article, list_categories
  resources/
    schema.py         # Resource definitions
  auth/
    middleware.py     # Token validation
  tests/

    conftest.py       # Test fixtures
  requirements.txt
  Dockerfile

Minimal Working Server — Hello World

python
"""
server.py — Minimal MCP server with one tool.
Test with: mcp dev server.py
"""
import asyncio
import json
from mcp.server import Server
from mcp.server.models import InitializationOptions
from mcp.types import Tool, TextContent, CallToolResult
import mcp.server.stdio

# Instantiate the server with a name — shown in /mcp output
server = Server("acme-knowledge-base")


@server.list_tools()
async def handle_list_tools() -> list[Tool]:
    """Return all tools this server exposes."""
    return [
        Tool(
            name="get_server_info",
            description="""Returns metadata about this MCP server:
            version, available tool count, and uptime.
            Use when the user asks about which MCP servers are connected
            or what version of the knowledge base server is running.""",
            inputSchema={"type": "object", "properties": {}}
        )
    ]


@server.call_tool()
async def handle_call_tool(name: str, arguments: dict) -> list[TextContent]:
    """Dispatch tool calls to handlers."""
    if name == "get_server_info":
        info = {
            "server": "acme-knowledge-base",
            "version": "1.0.0",
            "tools_available": 1,
            "status": "healthy"
        }
        return [TextContent(type="text", text=json.dumps(info, indent=2))]
    raise ValueError(f"Unknown tool: {name}")


async def main():
    async with mcp.server.stdio.stdio_server() as (read_stream, write_stream):
        await server.run(
            read_stream,
            write_stream,
            InitializationOptions(
                server_name="acme-knowledge-base",
                server_version="1.0.0",
                capabilities=server.get_capabilities(
                    notification_options=None,
                    experimental_capabilities={}
                )
            )
        )

if __name__ == "__main__":
    asyncio.run(main())

Designing and Implementing MCP Tools

The art of MCP tool design is writing schemas and descriptions that give Claude exactly enough information to use the tool correctly in every situation, without ambiguity, and without leaking implementation details it doesn't need.

Tool Naming Conventions

Use verb_noun naming: get_customer, search_documents, create_ticket, list_categories. Names must be unique within a server. Avoid generic names like get_data or process — Claude uses the name as the first signal for which tool to call.

Complete Example: Internal Knowledge Base Server (3 Tools)

python
"""
kb_server.py — Internal Knowledge Base MCP Server
Tools: search_kb, get_article, list_categories
Backend: Elasticsearch or simple JSON for demo
"""
import asyncio, json, os, httpx
from mcp.server import Server
from mcp.server.models import InitializationOptions
from mcp.types import Tool, TextContent
import mcp.server.stdio

server = Server("acme-knowledge-base")

KB_API_BASE = os.environ["KB_API_BASE"]    # e.g. https://kb.acme.internal/api
KB_API_TOKEN = os.environ["KB_API_TOKEN"]  # service account token

async def kb_get(path: str, params: dict = {}) -> dict:
    """Thin async wrapper around the KB REST API."""
    async with httpx.AsyncClient(timeout=15.0) as client:
        r = await client.get(
            f"{KB_API_BASE}{path}",
            params=params,
            headers={"Authorization": f"Bearer {KB_API_TOKEN}"}
        )
        r.raise_for_status()
        return r.json()


@server.list_tools()
async def list_tools() -> list[Tool]:
    return [
        Tool(
            name="search_kb",
            description="""Search the Acme internal knowledge base for articles, runbooks,
            architecture decisions, and process documentation.
            Use when the user asks about: internal processes, team conventions,
            deployment procedures, incident runbooks, architecture decisions (ADRs),
            or any company-specific knowledge not available in code.
            Returns articles ranked by relevance with title, snippet, category, and URL.
            Supports natural language queries — do not use boolean operators.""",
            inputSchema={
                "type": "object",
                "properties": {
                    "query": {
                        "type": "string",
                        "description": "Natural language search query (required)"
                    },
                    "category": {
                        "type": "string",
                        "description": "Filter by category slug (optional). Get slugs from list_categories.",
                        "enum": ["runbooks", "architecture", "processes", "security", "onboarding"]
                    },
                    "max_results": {
                        "type": "integer",
                        "description": "Number of results to return (default 5, max 20)",
                        "default": 5, "maximum": 20
                    }
                },
                "required": ["query"]
            }
        ),
        Tool(
            name="get_article",
            description="""Retrieve the full text of a specific knowledge base article by its ID.
            Use this after search_kb to get the complete content of a search result.
            Returns the full article: title, body, author, last_updated, tags, and related_articles.
            The article_id comes from the id field in search_kb results.""",
            inputSchema={
                "type": "object",
                "properties": {
                    "article_id": {
                        "type": "string",
                        "description": "Article ID from search_kb results (e.g. 'kb-001', 'arch-042')"
                    }
                },
                "required": ["article_id"]
            }
        ),
        Tool(
            name="list_categories",
            description="""List all available categories in the knowledge base with article counts.
            Use when the user wants to browse the knowledge base, understand what topics
            are covered, or before using search_kb with a category filter.
            Returns category slug, display name, description, and article count.""",
            inputSchema={
                "type": "object",
                "properties": {}
            }
        )
    ]


@server.call_tool()
async def call_tool(name: str, arguments: dict) -> list[TextContent]:
    try:
        if name == "search_kb":
            data = await kb_get("/search", {
                "q": arguments["query"],
                "category": arguments.get("category"),
                "limit": arguments.get("max_results", 5)
            })
            return [TextContent(type="text", text=json.dumps(data, indent=2))]

        elif name == "get_article":
            article_id = arguments["article_id"]
            # Input validation — never pass raw user input directly to path
            if not article_id.replace("-", "").replace("_", "").isalnum():
                raise ValueError(f"Invalid article_id format: {article_id!r}")
            data = await kb_get(f"/articles/{article_id}")
            return [TextContent(type="text", text=json.dumps(data, indent=2))]

        elif name == "list_categories":
            data = await kb_get("/categories")
            return [TextContent(type="text", text=json.dumps(data, indent=2))]

        else:
            raise ValueError(f"Unknown tool: {name}")

    except httpx.HTTPStatusError as e:
        # Never expose internal error details — return a safe error message
        error_msg = f"Knowledge base API error: HTTP {e.response.status_code}"
        return [TextContent(type="text", text=json.dumps({"error": error_msg}))]
    except httpx.TimeoutException:
        return [TextContent(type="text", text=json.dumps({"error": "Knowledge base request timed out"}))]
⚠
Always Validate Tool Input Before Using It
Claude constructs tool arguments from user input and its own reasoning. Always validate arguments before using them in API calls, database queries, or file paths. At minimum: check types, check string formats with allowlists or regex, and never construct URL paths or SQL using raw argument values without sanitisation.

Authentication Patterns for Custom Servers

Authentication in MCP servers follows different patterns depending on whether the server runs over stdio or HTTP. The right pattern also depends on whether you need per-user identity (for audit trails) or a shared service account identity (for simplicity).

┌─────────────────────────────────────────────────────────────────┐ │ MCP AUTHENTICATION PATTERNS │ └─────────────────────────────────────────────────────────────────┘ Pattern 1: stdio — Environment Variables (simplest) ────────────────────────────────────────────────── Claude Code config → env: { API_KEY: "${API_KEY}" } Server reads from os.environ["API_KEY"] at startup Secret stored in system keychain or secrets manager Identity: shared service account for all users on this machine Pattern 2: HTTP+SSE — Bearer Token (enterprise) ─────────────────────────────────────────────── Client sends: Authorization: Bearer <user-token> Server validates token against IdP (Okta, Azure AD, etc.) Extracts user_id, roles from claims Identity: per-user (enables per-user audit trails) Pattern 3: HTTP+SSE — OAuth 2.0 (full enterprise SSO) ────────────────────────────────────────────────────── Claude Code obtains OAuth token from IdP Sends token in Authorization header to MCP server Server introspects token at IdP's /token_info endpoint Full SSO integration — token rotation automatic

FastAPI MCP Server with Bearer Token Validation

python
"""
http_kb_server.py — HTTP+SSE MCP server with Bearer token auth.
Runs as a shared enterprise service, not a local stdio process.
"""
import os, json, hmac, secrets
from fastapi import FastAPI, Request, HTTPException, Depends
from fastapi.responses import StreamingResponse
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials

app = FastAPI(title="Acme Knowledge Base MCP Server")
security = HTTPBearer()

# In production: validate against IdP (Okta, Azure AD, internal auth service)
# For demonstration: compare against pre-shared token list
VALID_TOKENS = os.environ.get("MCP_VALID_TOKENS", "").split(",")

async def validate_bearer_token(
    credentials: HTTPAuthorizationCredentials = Depends(security)
) -> str:
    """Validate Bearer token and return the user identity."""
    token = credentials.credentials

    # Use constant-time comparison to prevent timing attacks
    valid = any(
        hmac.compare_digest(token, valid_token)
        for valid_token in VALID_TOKENS
        if valid_token
    )
    if not valid:
        raise HTTPException(
            status_code=401,
            detail="Invalid or expired token",
            headers={"WWW-Authenticate": "Bearer"}
        )
    # In production: decode JWT and return user_id from claims
    return f"user-{token[:8]}"


@app.post("/message")
async def handle_message(request: Request, user_id: str = Depends(validate_bearer_token)):
    """Handle JSON-RPC messages from the MCP client."""
    body = await request.json()
    method = body.get("method")
    params = body.get("params", {})
    request_id = body.get("id")

    # Log every tool call for audit trail
    if method == "tools/call":
        print(json.dumps({
            "event": "tool_call",
            "user_id": user_id,
            "tool": params.get("name"),
            "arguments": params.get("arguments", {})
        }))

    result = await dispatch_mcp_method(method, params, user_id)
    return {"jsonrpc": "2.0", "id": request_id, "result": result}


@app.get("/health")
async def health_check():
    return {"status": "healthy", "server": "acme-knowledge-base"}
ℹ
Service Account vs Per-User Identity
For most enterprise MCP servers, a service account (shared identity) is sufficient and far simpler to manage. Use per-user identity only when you need fine-grained audit logs that attribute each tool call to a specific person, or when different users should see different data subsets. Per-user identity requires a proper OAuth integration with your IdP.

Deployment Models

Choosing the right deployment model for your MCP server depends on who needs access to it, what security posture you need, and how much operational complexity you can absorb. There are three primary models, each with clear trade-offs.

ModelBest ForComplexitySecurity SurfaceScaling
stdio / Local Dev tools, personal productivity, small team (1-5 people) Minimal — just a Python file Process isolation. Auth via env vars. No network exposure. Scales with the developer's machine. No concurrent users.
Docker Container Reproducible team tooling, CI/CD environments, staging Medium — Dockerfile + compose Container isolation. Secrets as env vars at runtime. No host access. Run multiple containers. No built-in load balancing.
Cloud-Hosted (Lambda / Cloud Run) Enterprise-wide shared servers, high availability, >50 users High — CI/CD, secrets manager, TLS, monitoring Full network controls. IAM roles. TLS 1.3. WAF. Audit logs. Auto-scales to thousands of concurrent connections.

Dockerfile for a Production Python MCP Server

dockerfile
# Multi-stage build — keeps final image small
FROM python:3.12-slim AS builder

WORKDIR /app
COPY requirements.txt .
RUN pip install --no-cache-dir --user -r requirements.txt

# Final stage — no build tools in production image
FROM python:3.12-slim AS runtime

# Run as non-root user — never run MCP servers as root
RUN useradd --uid 1001 --no-create-home --shell /sbin/nologin mcp
USER mcp

WORKDIR /app

# Copy installed packages from builder
COPY --from=builder /root/.local /home/mcp/.local
ENV PATH=/home/mcp/.local/bin:$PATH

# Copy application code
COPY --chown=mcp:mcp . .

# Health check — used by ECS, Kubernetes, Cloud Run
HEALTHCHECK --interval=30s --timeout=5s --start-period=10s --retries=3 \
  CMD python3 -c "import httpx; httpx.get('http://localhost:8080/health').raise_for_status()"

EXPOSE 8080

CMD ["python3", "-m", "uvicorn", "http_kb_server:app", \
     "--host", "0.0.0.0", "--port", "8080", "--workers", "2"]

Testing MCP Servers

MCP servers are just Python functions under the decorator layer. This makes them straightforward to unit test — no Claude needed, no MCP protocol overhead. Test the tool handler functions directly with mocked dependencies.

python
"""
tests/test_tools.py — Unit tests for KB MCP server tool handlers.
Run with: pytest tests/ -v
"""
import json
import pytest
from unittest.mock import AsyncMock, patch
from mcp.types import TextContent

# Import the call_tool handler from your server module
from kb_server import call_tool


# ── Test fixtures ──────────────────────────────────────────────────────────────

@pytest.fixture
def mock_search_response():
    return {
        "results": [
            {"id": "kb-042", "title": "Deployment Runbook",
             "snippet": "Steps to deploy to production...",
             "category": "runbooks", "score": 0.95}
        ],
        "total": 1
    }


# ── search_kb tests ────────────────────────────────────────────────────────────

@pytest.mark.asyncio
async def test_search_kb_happy_path(mock_search_response):
    """search_kb returns results for a valid query."""
    with patch("kb_server.kb_get", new_callable=AsyncMock) as mock_get:
        mock_get.return_value = mock_search_response

        result = await call_tool("search_kb", {"query": "deployment runbook"})

        assert len(result) == 1
        assert isinstance(result[0], TextContent)
        data = json.loads(result[0].text)
        assert data["total"] == 1
        assert data["results"][0]["id"] == "kb-042"

        # Verify the correct API endpoint was called
        mock_get.assert_called_once_with("/search", {
            "q": "deployment runbook",
            "category": None,
            "limit": 5
        })


@pytest.mark.asyncio
async def test_search_kb_with_category_filter(mock_search_response):
    """search_kb passes category filter correctly."""
    with patch("kb_server.kb_get", new_callable=AsyncMock) as mock_get:
        mock_get.return_value = mock_search_response
        await call_tool("search_kb", {"query": "deploy", "category": "runbooks"})
        args = mock_get.call_args[0]
        assert args[1]["category"] == "runbooks"


# ── get_article tests ──────────────────────────────────────────────────────────

@pytest.mark.asyncio
async def test_get_article_rejects_path_traversal():
    """get_article must reject article_ids with path traversal characters."""
    result = await call_tool("get_article", {"article_id": "../../../etc/passwd"})
    data = json.loads(result[0].text)
    assert "error" in data


@pytest.mark.asyncio
async def test_unknown_tool_raises_value_error():
    """Unknown tool names must raise ValueError, not silently fail."""
    with pytest.raises(ValueError, match="Unknown tool: nonexistent_tool"):
        await call_tool("nonexistent_tool", {})


# ── Schema validation tests ────────────────────────────────────────────────────

def test_all_tool_schemas_are_valid_json_schema():
    """All tool inputSchemas must be valid JSON Schema objects."""
    import jsonschema
    from kb_server import list_tools
    import asyncio
    tools = asyncio.run(list_tools())
    for tool in tools:
        # This raises if the schema is not valid JSON Schema
        jsonschema.Draft7Validator.check_schema(tool.inputSchema)
        assert tool.name, f"Tool missing name: {tool}"
        assert len(tool.description) > 50, \
            f"Tool '{tool.name}' has a description that is too short to be useful"

Common MCP Server Bugs to Test For
  • Wrong JSON Schema types (Python int vs JSON Schema "integer")
  • Missing required field list in inputSchema
  • Unhandled exceptions that leak internal stack traces
  • Path traversal in any argument used to construct file/URL paths
  • Tool descriptions shorter than ~50 characters (too vague for Claude)
